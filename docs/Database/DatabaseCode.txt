//// -- LEVEL 1
//// -- Tables and References

// Creating tables
Table User {
  id int [PK]
  email String
  password String
  name String
  surname String
  faculty String
  type String
}

Table rooms {
  room_name String PK
  building building PK
  faculty String
  faculty_specific boolean
  nPeople int
  screen boolean
  plugs int
  beamer boolean
}

Table building {
  name String
  street String
  house_number int
  building_number int PK
}

Table building_hours {
  building_number building_number PK
  day int
  open_time time
  close_time time
}

Table restaurant_hours {
  restaurant restaurant PK
  day int
  open_time time
  close_time time
}

Table bikes {
  id int PK
  location building_number
  available boolean
}

Table bike_reservations {
  id id PK
  from datetime
  to datetime
  from_building building_number
  to_building building_number
  User User
}

Table room_reservations {
  id id PK
  from datetime
  to datetime
  User User
}

Table restaurant {
  id int PK
  building building_number
  name String
}

Table menu {
  restaurant id PK
  menu id PK
  item String
  item_ingredients String
  item_allergies String
}

Table food_orders {
  id int PK
  User User
  restaurant restaurant
  delivery_time int
  delivery_date date
  delivery_location building_number
}

Table room_feedback {
  feedback_id int PK
  client User
  recipient User
  room_reservations room_reservations
  feedback_date date
  feedback_text String
}

Table food_feedback {
  feedback_id int PK
  client User
  recipient User
  food_orders food_orders
  feedback_date date
  feedback_text String
}

Table bike_feedback {
  feedback_id int PK
  client User
  recipient User
  bike_reservations bike_reservations
  feedback_date date
  feedback_text String
}

Ref: "User"."id" < "food_orders"."User"

Ref: "User"."id" < "bike_reservations"."User"

Ref: "restaurant"."id" < "restaurant_hours"."restaurant"

Ref: "bikes"."id" < "bike_reservations"."id"

Ref: "bikes"."location" < "building"."building_number"

Ref: "building"."building_number" < "restaurant"."building"

Ref: "restaurant"."id" < "food_orders"."restaurant"

Ref: "building"."building_number" < "food_orders"."delivery_location"

Ref: "rooms"."room_name" < "room_reservations"."id"

Ref: "room_reservations"."User" < "User"."id"
